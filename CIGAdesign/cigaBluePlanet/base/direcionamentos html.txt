document.addEventListener("DOMContentLoaded", function() {
    const video = document.getElementById("ciga-promo-video");
    const unmuteBtn = document.getElementById("ciga-unmute-btn");

    // Prepara o vídeo para tocar com volume no máximo
    video.volume = 1;
    video.muted = false; // Tenta forçar o estado desmutado

    // Inicia o play e captura a promessa (comportamento nativo do navegador)
    const playPromise = video.play();

    if (playPromise !== undefined) {
        playPromise.then(() => {
            // Sucesso! O navegador permitiu o Autoplay COM som.
            console.log("Autoplay com som iniciado com sucesso.");
        }).catch(error => {
            // Falha: O navegador bloqueou o som por falta de interação prévia.
            console.log("Autoplay com som bloqueado. Entrando em modo 'Mudo' para não perder a animação.");
            
            // Reverte para mudo e toca de novo (isso o navegador sempre permite)
            video.muted = true;
            video.play();
            
            // Mostra o botão para o usuário ativar a experiência sonora do presskit
            unmuteBtn.style.display = "inline-block";
        });
    }

    // Ação de clique no botão de som
    unmuteBtn.addEventListener("click", () => {
        if(video.muted) {
            video.muted = false;
            unmuteBtn.innerText = "Desativar Som 🔇";
        } else {
            video.muted = true;
            unmuteBtn.innerText = "Ativar Som 🔊";
        }
    });
});
